Table Commands {
  id integer [primary key]
  name varchar
  description varchar
  created_at timestamp
  updated_at timestamp
}

Table Users {
  id integer [primary key]
  first_name varchar
  last_name varchar
  nick_name varchar [unique]
  email varchar
  password varchar
  status varchar
  created_at timestamp
  updated_at timestamp
}

Table User_Channel {
  id int [primary key]
  user_id integer
  channel_id integer
  role varchar
  notification_settings varchar
  kick_count integer
  created_at timestamp
  updated_at timestamp
}

Table Channel {
  id integer [primary key]
  name varchar [unique]
  type varchar
  created_by integer [not null]
  last_active_at timestamp
  created_at timestamp
  updated_at timestamp
}

Table Channel_History {
  id integer [primary key]
  channel_id integer
  last_fetched_message_id integer
  has_more varchar
  created_at timestamp
  updated_at timestamp
}

Table Message {
  id integer [primary key]
  channel_id integer [not null]
  sender_id integer [not null]
  content varchar
  has_ping bool
  has_command bool
  sent_at timestamp
  created_at timestamp
  updated_at timestamp
}

Table Message_Mentions {
  id integer [primary key]
  mentioned_user_id integer [not null]
  message_id integer [not null]
  created_at timestamp
  updated_at timestamp
}

Table User_Message_Commands {
  id integer [primary key]
  message_id integer [not null]
  command_id integer [not null]
  user_id integer [not null]
  args varchar
  created_at timestamp
  updated_at timestamp
}
