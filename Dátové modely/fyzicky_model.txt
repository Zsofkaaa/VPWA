Table Commands {
  id integer [primary key]
  name varchar
  description varchar
  created_at timestamp
  updated_at timestamp
}

Table User_Commands {
  id integer [primary key]
  user_id integer
  command_id integer
  message_id integer
  object varchar
  created_at timestamp
  updated_at timestamp
}

Table Users {
  id integer [primary key]
  first_name varchar
  last_name varchar
  nick_name varchar [unique]
  email varchar
  password varchar
  status varchar
  created_at timestamp
  updated_at timestamp
}

Table User_Channel {
  id int [primary key]
  user_id integer
  channel_id integer
  role varchar
  notification_settings varchar
  kick_count integer
  created_at timestamp
  updated_at timestamp
}

Table Channel {
  id integer [primary key]
  name varchar [unique]
  type varchar
  created_by integer [not null]
  last_active_at timestamp
  created_at timestamp
  updated_at timestamp
}

Table Channel_History {
  id integer [primary key]
  channel_id integer
  last_fetched_message_id integer
  has_more varchar
  created_at timestamp
  updated_at timestamp
}

Table Message {
  id integer [primary key]
  channel_id integer [not null]
  sender_id integer [not null]
  content varchar
  sent_at timestamp
  created_at timestamp
  updated_at timestamp
}



Ref: User_Commands.command_id > Commands.id
Ref: User_Commands.user_id > Users.id
Ref: User_Commands.message_id - Message.id
Ref: Message.sender_id > Users.id
Ref: Message.channel_id > Channel.id
Ref: Channel_History.last_fetched_message_id - Message.id
Ref: Channel_History.channel_id - Channel.id
Ref: Channel.created_by > Users.id
Ref: User_Channel.user_id > Users.id
Ref: User_Channel.channel_id > Channel.id
